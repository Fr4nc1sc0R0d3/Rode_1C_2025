179void controlarBuzzerYMotores() {
180 bool seSuperoUmbralSuperior = false;
181 bool seSuperoUmbralInferior = false;
182
183 for (int i = 0; i < 3; i++) {
184 if (i == 0) {
185 if (medicionesSensores[i] <= umbralSuperior) {
186 seSuperoUmbralInferior = true;
187 break;
188 }
189 } else {
190 if (medicionesSensores[i] <= umbralInferior) {
191 seSuperoUmbralSuperior = true;
192 break;
193 }
194 }
195 }
196
197 if (seSuperoUmbralSuperior) {
198 GPIOOn(GPIO_12);
199 GPIOOn(GPIO_18);
200 } else {
201 GPIOOff(GPIO_12);
202 }
203
204 if (seSuperoUmbralInferior) {
205 GPIOOn(GPIO_20);
206 GPIOOn(GPIO_18);
207 } else {
208 GPIOOff(GPIO_20);
209 }
210
211 if (!seSuperoUmbralInferior && !seSuperoUmbralSuperior) {
212 GPIOOff(GPIO_18);
213 GPIOOff(GPIO_12);
214 GPIOOff(GPIO_20);
215 }
216}

160static void tareaMedirRadiacion(void *pvParameter){
161
162 while(true)
163 {
164 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
165 if (encendido) {
166 char bufferR[15];
167 AnalogInputReadSingle(CANAL_INGRESO_SENAL, &voltaje);
168 sprintf(bufferR, "%.2f", (float)voltaje);
169 UartSendString(UART_PC, "Radiación: ");
170 UartSendString(UART_PC, bufferR);
171 UartSendString(UART_PC, "mR/h");
172 if (devolverRadiacionSensada(voltaje) > 40) {
173 radiacionAlta = true;
174 LedOn(LED_2);
175 UartSendString(UART_PC, " - RADIACIÓN ELEVADA");
176 }
177 radiacionAlta = false;
178 UartSendString(UART_PC, " \r\n");
179 }
